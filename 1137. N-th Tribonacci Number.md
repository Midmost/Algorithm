```python3
class Solution:
    def tribonacci(self, n: int) -> int:
        #Generalization:
        #f(n) = T_n
        
        #base: 
        #f(0) = 0, f(1) = 1, f(2) = 1
        
        
        #general:
        # n >= 0, T_n+3 = T_n + T_n+1 + T_n+2
        
        cache = [-1] * (n+1)
        
        #topdown 함수를 만들 때 파라미터 값이 헷갈림;;
        def topdown(k):
            if k == 0 or k == 1:
                return k
            if k == 2:
                return 1
            if cache[k] is not -1 and k >= 0:
                return cache[k]
            cache[k + 3] = topdown(k) + topdown(k+1) + topdown(k+2) 
            return cache[k]
        return topdown(n)
            
        
        
        

```
